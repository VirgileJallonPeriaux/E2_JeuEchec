@name Calcul trajectoir
@persist E:entity
interval(100)
if(first()|duped()){holoCreate(1) E=owner()}
if(E:pos():z()>=5){

    G=gravity()
    
    Hauteur=E:pos():z()
    
    VelVer=E:vel():z()
    VelHor=E:vel():x()
    Xr=(VelHor/G)*(VelVer+sqrt(VelVer^2+2*G*Hauteur))

    VelVer=E:vel():z()
    VelHor=E:vel():y()
    Yr=(VelHor/G)*(VelVer+sqrt(VelVer^2+2*G*Hauteur))
    
    Vec=E:pos()+vec(Xr,Yr,-E:pos():z())
    holoAng(1,ang(0,0,0))
    holoScale(1,vec(1))
    holoPos(1,Vec)    
}
